document.addEventListener("DOMContentLoaded", function() {
  const btn = document.getElementById("showPreviousPublicationsBtn");
  const marker = document.getElementById("previous-publications-toggle-end");
  const previousHeading = Array.from(document.querySelectorAll("h2")).find(function(heading) {
    return heading.textContent.trim() === "Previous years";
  });
  if (!btn || !previousHeading || !marker) {
    return;
  }
  const previousNodes = [];
  let node = previousHeading;
  while (node && node !== marker) {
    previousNodes.push(node);
    node = node.nextElementSibling;
  }
  function setPreviousPublicationsVisible(isVisible) {
    previousNodes.forEach(function(previousNode) {
      previousNode.classList.toggle("is-hidden", !isVisible);
    });
    btn.textContent = isVisible ? "Hide Previous years" : "Show Previous years";
  }
  setPreviousPublicationsVisible(false);
  btn.addEventListener("click", function() {
    setPreviousPublicationsVisible(previousHeading.classList.contains("is-hidden"));
  });
});
